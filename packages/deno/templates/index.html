First Steps

Register a request handler.

import stewpot from 'stewpot/stewpot.js';

export function main() {
  return stewpot({
    handler({ pathname }) {
      if (pathname === "/") {
        return () => new Response("Hello World!");
      }
    },
  });  
}

if (import.meta.main) {
  main();
}

Next Steps

Render the index template.

import stewpot from "stewpot/stewpot.js";

export function main() {
  return stewpot({
    handler({ pathname, render }) {
      if (pathname === "/") {
        return async () => new Response(await render('index'), { headers: { 'content-type': 'text/html' } });
      }
    },
  });
}

if (import.meta.main) {
  main();
}

Further Steps

Take a look at the examples on GitHub for getting started. Make sure to check out the default request handler, which is responsible for rendering this very page.
