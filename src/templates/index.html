import { stewpot, useRouter } from 'stewpot';

// 1. create app and router instances
const app = stewpot();
const router = useRouter();

// 2. define root (/) path
router.get('root', (_, response) => {
  response.writeHead(200, { 'Content-Type': 'text/html' });
  response.end('<h1>Hello World!</h1>');
});

// 3. register request handler
app.use((_, response) => {
  try {
    return router.route(_, response);
  } catch(err) {
    console.log(err);
    response.end();
  }
});
